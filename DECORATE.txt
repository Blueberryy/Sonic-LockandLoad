ACTOR SonicPlayer : PlayerPawn
{
	Speed 2
	Health 125
	Radius 10
	Height 46
	PainChance 255
	Player.MaxHealth 999
	Player.DisplayName "Sonic the Hedgehog"
	Player.JumpZ 10.0
	Player.StartItem "Boost"
	Player.StartItem "ChaosEnergy", 300
	States
  {
  Spawn:
    PLAY A -1
    Loop
  See:
    PLAY ABCD 4 
    Loop
  Missile:
    PLAY E 12
    Goto Spawn
  Melee:
    PLAY F 6 BRIGHT
    Goto Missile
  Pain:
    PLAY G 4 
    PLAY G 4 A_Pain
    Goto Spawn
  Death:
    PLAY H 0 A_PlayerSkinCheck("AltSkinDeath")
  Death1:
    PLAY H 10
    PLAY I 10 A_PlayerScream
    PLAY J 10 A_NoBlocking
    PLAY KLM 10
    PLAY N -1
    Stop
  XDeath:
    PLAY O 0 A_PlayerSkinCheck("AltSkinXDeath")
  XDeath1:
    PLAY O 5
    PLAY P 5 A_XScream
    PLAY Q 5 A_NoBlocking
    PLAY RSTUV 5
    PLAY W -1
    Stop
  AltSkinDeath:
    PLAY H 6
    PLAY I 6 A_PlayerScream
    PLAY JK 6
    PLAY L 6 A_NoBlocking
    PLAY MNO 6
    PLAY P -1
    Stop
  AltSkinXDeath:
    PLAY Q 5 A_PlayerScream
    PLAY R 0 A_NoBlocking
    PLAY R 5 A_SkullPop
    PLAY STUVWX 5
    PLAY Y -1
    Stop
  }
}

ACTOR "Boost" : Weapon replaces Fist 1220
{
  Weapon.SelectionOrder 3700
  Weapon.Kickback 500
  Weapon.AmmoType "ChaosEnergy"
  +WEAPON.MELEEWEAPON
  States
  {
  Ready:
    PUNG A 1 A_WeaponReady
    Loop
  Deselect:
    TNT1 AA 0 A_Lower
	PUNG A 1 A_Lower
	Goto Select+1
  Select:
    TNT1 AA 0 A_Lower
	PUNG A 1 A_Lower
	Goto Deselect+1
  Fire:
    PUNG A 0 A_PlaySound("sonicdoom/boost1")
	PUNG A 0 A_TakeInventory("ChaosEnergy",15)
	PUNG A 1 A_Recoil(-64)
    PUNG D 10 A_CustomPunch(200, false)
    Goto Ready
  AltFire:
    PUNG A 0 A_PlaySound("sonicdoom/drift")
    PUNG A 0 A_Recoil(-8)
    PUNG B 1 A_CustomPunch(1)
    Goto AfterDrift
  AfterDrift:
	PUNG A 0 A_PlaySound("sonicdoom/aboost")
	Goto Ready
  }
}
