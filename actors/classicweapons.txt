ACTOR SonicFist : Weapon replaces Fist 1212
{
  Weapon.SelectionOrder 3700
  Weapon.Kickback 10
  Tag "Sonic's Fist"
  +WEAPON.MELEEWEAPON
  States
  {
  Ready:
    PUNG A 1 A_WeaponReady
    Loop
  Deselect:
    PUNG A 1 A_Lower
    Loop
  Select:
    PUNG A 1 A_Raise
    Loop
  Fire:
    PUNG B 1
    PUNG C 1 A_Punch
    PUNG D 1
    PUNG C 1
    PUNG B 1 A_ReFire
    Goto Ready
  AltFire:
    TNT1 A 0 A_JumpIf(CountInv("PowerInvulnerable") < 1, "Drift")
    TNT1 A 0 A_JumpIf(CountInv("PowerInvulnerable") > 0, "SuperDrift")
  Drift:
    TNT1 A 0 A_PlaySound("sonicdoom/drift", 2)
    TNT1 A 0 A_Recoil(-6)
	TNT1 A 0 A_Light(2)
	TNT1 A 0 A_GiveInventory("ChaosEnergy")
    SPIN A 1 A_CustomPunch(2)
    TNT1 A 0 A_ReFire
    TNT1 A 0 A_StopSound(2)
	TNT1 A 0 A_Light(0)
	Goto Ready
  }
}

ACTOR ChaosBlaster : Weapon
{
  Weapon.SelectionOrder 100
  Weapon.AmmoUse 2
  Weapon.AmmoGive 25
  Weapon.KickBack 100
  Weapon.AmmoType "ChaosEnergy"
  Tag "Chaos Blaster"
  Inventory.PickupMessage "Picked up the Chaos energy blaster!"
  States
  {
  Ready:
    PLSG A 1 A_WeaponReady
    Loop
  Deselect:
    PLSG A 1 A_Lower
    Loop
  Select:
    PLSG A 1 A_Raise
    Loop
  Fire:
    PLSG A 0 A_FirePlasma
	PLSG A 1
	{
		if (CountInv("PowerSuper") > 0)
		{
			A_GiveInventory("ChaosEnergy", 2);
		}
	}
    Goto Ready
  Flash:
    PLSF A 4 Bright A_Light1
    Goto LightDone
    PLSF B 4 Bright A_Light1
    Goto LightDone
  Spawn:
    PLAS A -1
    Stop
  }
}
