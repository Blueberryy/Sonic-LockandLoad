ACTOR Ring : Health replaces HealthBonus 1000
{
  Inventory.PickupMessage "Picked up a ring."
  Inventory.PickupSound "sonicdoom/ring"
  Inventory.MaxAmount 999
  +COUNTITEM
  +INVENTORY.ALWAYSPICKUP
  States
  {
  Spawn:
    RIN1 A -1
    Stop
  }
}
