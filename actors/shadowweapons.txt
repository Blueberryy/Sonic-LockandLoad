ACTOR ShadowAbilities : Weapon
{
  Weapon.SelectionOrder 3700
  Weapon.Kickback 9999999999999
  Weapon.BobRangeX 0.0
  Weapon.BobRangeY 0.0
  Weapon.AmmoType "ChaosEnergy"
  Tag "Shadow the Hedgehog"
  +WEAPON.MELEEWEAPON
  +WEAPON.NOAUTOFIRE
  States
  {
  Ready:
    TNT1 A 1 A_WeaponReady
    Loop
  Deselect:
    TNT1 AA 0 A_Lower
	Goto Select
  Select:
	TNT1 A 1 A_Lower
	Goto Deselect
  Fire:
    TNT1 A 0 A_JumpIf(CountInv("PowerSuper") > 0, "SuperKick")
    TNT1 A 0 A_PlaySound("sonicdoom/swordswing",1)
    DKIK ABC 1
    DKIK D 1 A_CustomPunch(random(1,16)*10, true, CPF_NOTURN, "NoGibPuff", 75, 0, 0, "ArmorBonus", "sonicdoom/finalhit")
    DKIK EF 1
	TNT1 A 2 A_ReFire
    Goto Ready
  SuperKick:
    TNT1 A 0 A_PlaySound("sonicdoom/swordswing",1)
    SDKI BC 1
	SDKI D 1 A_CustomPunch(random(1,16)*100, true, CPF_NOTURN, "GibPuff", 75, 0, 0, "ArmorBonus", "sonicdoom/finalhit")
    SDKI EF 1
	TNT1 A 2
    Goto Ready
  AltFire:
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergy") < 2, "NoEnergy")
    SARM ABC 1
    SARM D 6
	{
	  if (CountInv("PowerSuper")) {
        A_FireProjectile("ChaosSpear");
	  }
	  else {
	    A_FireProjectile("ChaosSpear");
		A_TakeInventory("ChaosEnergy", 2);
	  }
	}
	SARM D 0 A_ReFire
    SARM CBA 1
	Goto Ready
  AltHold:
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergy") < 2, "NoEnergy2")
    SARM D 6
	{
	  if (CountInv("PowerSuper")) {
        A_FireProjectile("ChaosSpear");
	  }
	  else {
	    A_FireProjectile("ChaosSpear");
		A_TakeInventory("ChaosEnergy", 2);
	  }
	}
	SARM A 0 A_ReFire
	Goto AltFire
  NoEnergy:
    TNT1 A 0 A_Print("\c[ShadowRedBlack]I don't have enough Chaos Energy!")
	TNT1 A 0 A_PlaySound("sonicdoom/shadowerror", 4096)
	Goto Ready
  NoEnergy2:
    SARM CBA 1
    TNT1 A 0 A_Print("\c[ShadowRedBlack]I don't have enough Chaos Energy!")
	TNT1 A 0 A_PlaySound("sonicdoom/shadowerror", 4096)
	Goto Ready
  Spawn:
    SHAD A -1
    Stop
  }
}

ACTOR ChaosEmerald : Weapon
{
  Weapon.SelectionOrder 3700
  Weapon.AmmoType "ChaosEnergyEX"
  Weapon.AmmoUse 25
  Weapon.AmmoGive 100
  Inventory.PickupSound "sonicdoom/damn4thchaos"
  Inventory.PickupMessage "\c[ShadowRedBlack]The damn 4th Chaos Emerald!"
  Tag "Chaos Emerald"
  +AMMO_OPTIONAL
  States
  {
  Ready:
    CARM A 1 A_WeaponReady
    Loop
  Deselect:
    CARM AA 0 A_Lower
	Goto Select
  Select:
	CARM A 1 A_Lower
	Goto Deselect
  Fire:
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergyEX") < 25, "NotEnoughEnergy")
	TNT1 A 0 A_PlaySound("sonicdoom/chaosgrunt", 2)
    CARM A 1 A_FireProjectile("SuperChaosEnergyBall")
	CARM BCDE 1
	CARM E 10
	CARM DCB 1
    Goto Ready
  AltFire:
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergyEX") > 0, "EnergyCheck")
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergyEX") < 1, "NoEnergy")
  EnergyCheck:
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergy") > 998, "EnergyFull")
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergy") < 999, "ChaosCharge")
  NotEnoughEnergy:
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergyEX") < 1, "NoEnergy")
    TNT1 A 0 A_Print("\c[ShadowRedBlack]I don't have enough emerald energy!")
	TNT1 A 0 A_PlaySound("sonicdoom/shadowerror", 4096)
	Goto Ready
  NoEnergy:
    TNT1 A 0 A_Print("\c[ShadowRedBlack]My Emerald is out of energy!")
	TNT1 A 0 A_PlaySound("sonicdoom/shadowerror", 4096)
	Goto Ready
  EnergyFull:
    TNT1 A 0 A_Print("\c[ShadowRedBlack]My Chaos energy is full!")
	TNT1 A 0 A_PlaySound("sonicdoom/shadowerror", 4096)
	Goto Ready
  ChaosCharge:
    TNT1 A 0 A_PlaySound("sonicdoom/healthcharge", 4096)
    CARM F 2 A_TakeInventory("ChaosEnergyEX", 1)
    TNT1 A 0 A_GiveInventory("ChaosEnergy", 5)
	TNT1 A 0 A_ReFire
	Goto Ready
  Spawn:
    CHSM B -1
    Stop
  }
}

ACTOR ShadowEclipseBlaster : Weapon
{
  Weapon.SelectionOrder 1400
  Weapon.AmmoType1 "ChaosEnergy"
  Weapon.AmmoType2 "ChaosEnergy"
  Weapon.AmmoUse 5
  Weapon.AmmoGive 30
  Weapon.KickBack 100000000
  Weapon.BobRangeX 0.0
  Weapon.BobRangeY 1.0
  Weapon.BobSpeed 1.0
  +EXTREMEDEATH
  Inventory.PickupMessage "\c[ShadowRedBlack]The sixth model of the Eclipse Blaster. Four streams means more hits."
  Tag "Eclipse Blaster Mk. VI"
  States
  {
  Ready:
    ECBL A 1 A_WeaponReady
    Loop
  Deselect:
    TNT1 AA 0 A_Lower
	ECBL A 1 A_Lower
	Goto Select+1
  Select:
    TNT1 AA 0 A_Lower
	ECBL A 1 A_Lower
	Goto Deselect+1
  Fire:
	ECBL B 1
	{
	  if (CountInv("PowerSuper")) {
        A_FireProjectile("ChaosEnergyBall", 0, 0, -8, -8);
        A_FireProjectile("SilentChaosEnergyBall", 0, 0, 8, 6);
        A_FireProjectile("SilentChaosEnergyBall", 0, 0, -8, 6);
        A_FireProjectile("SilentChaosEnergyBall", 0, 0, 8, -8);
	  }
	  else {
        A_FireProjectile("ChaosEnergyBall", 0, 1, -8, -8);
        A_FireProjectile("SilentChaosEnergyBall", 0, 1, 8, 6);
        A_FireProjectile("SilentChaosEnergyBall", 0, 1, -8, 6);
        A_FireProjectile("SilentChaosEnergyBall", 0, 1, 8, -8);
	  }
	}
    Goto Ready
  AltFire:
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergy") > 0, "RingCheck")
    TNT1 A 0 A_JumpIf(CountInv("ChaosEnergy") < 1, "NoEnergy")
  RingCheck:
    TNT1 A 0 A_JumpIf(Health > 998, "RingsFull")
    TNT1 A 0 A_JumpIf(Health < 999, "ChaosHeal")
  NoEnergy:
    TNT1 A 0 A_Print("\c[ShadowRedBlack]My energy is depleted!")
	TNT1 A 0 A_PlaySound("sonicdoom/shadowerror", 4096)
	Goto Ready
  RingsFull:
    TNT1 A 0 A_Print("\c[ShadowRedBlack]I'm already at full Rings!")
	TNT1 A 0 A_PlaySound("sonicdoom/shadowerror", 4096)
	Goto Ready
  ChaosHeal:
    ECBL C 0 A_PlaySound("sonicdoom/healthcharge", 4096)
    ECBL C 2 A_TakeInventory("ChaosEnergy", 1)
    ECBL C 0 A_GiveInventory("Health", 8)
	Goto Ready
  Spawn:
    ECLI A -1
    Stop

  }
}
